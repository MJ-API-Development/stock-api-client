  import requests
  url = 'https://gateway.eod-stock-api.site/api/v1/exchanges'
  headers = {
      'Content-Type': 'application/json'
  }

  params = {'api_key':'API KEY HERE'}
  response = requests.get(url, headers=headers, params=params)
  print(response.json())
